#include <torch/torch.h>
#include <vector>

struct Experience {
    torch::Tensor state;
    int64_t action;
    float reward;
    torch::Tensor nextState;
    bool done;
};

class DQNImpl : public torch::nn::Module {
public:
    DQNImpl(int64_t inputSize, int64_t hiddenSize, int64_t outputSize)
        : fc1(register_module("fc1", torch::nn::Linear(inputSize, hiddenSize))),
          fc2(register_module("fc2", torch::nn::Linear(hiddenSize, outputSize))) {}

    torch::Tensor forward(const torch::Tensor& x) {
        return fc2->forward(torch::relu(fc1->forward(x)));
    }
private:
    torch::nn::Linear fc1{nullptr}, fc2{nullptr};
};
TORCH_MODULE(DQN);

void trainDQN(DQN& policyNet,
              DQN& targetNet,
              torch::optim::Adam& optimizer,
              const std::vector<Experience>& replayBuffer,
              int64_t batchSize,
              float gamma,
              torch::Device device)
{
    if (replayBuffer.size() < batchSize) return;

    std::vector<Experience> batch;
    std::sample(replayBuffer.begin(), replayBuffer.end(), std::back_inserter(batch),
                batchSize, std::mt19937{std::random_device{}()});

    std::vector<torch::Tensor> states, nextStates;
    std::vector<int64_t> actions;
    std::vector<float> rewards;
    std::vector<bool> dones;

    for (const auto& e : batch) {
        states.push_back(e.state.to(device));
        nextStates.push_back(e.nextState.to(device));
        actions.push_back(e.action);
        rewards.push_back(e.reward);
        dones.push_back(e.done);
    }

    auto stateBatch = torch::stack(states);
    auto nextStateBatch = torch::stack(nextStates);
    auto actionBatch = torch::tensor(actions, torch::kLong).to(device);
    auto rewardBatch = torch::tensor(rewards).to(device);
    auto doneBatch = torch::tensor(dones).to(device);

    auto qValues = policyNet->forward(stateBatch).gather(1, actionBatch.unsqueeze(1));
    auto nextQValues = targetNet->forward(nextStateBatch).max(1).values.detach();
    auto expectedQValues = rewardBatch + (gamma * nextQValues * (1 - doneBatch));

    auto loss = torch::mse_loss(qValues, expectedQValues.unsqueeze(1));

    optimizer.zero_grad();
    loss.backward();
    optimizer.step();
}
