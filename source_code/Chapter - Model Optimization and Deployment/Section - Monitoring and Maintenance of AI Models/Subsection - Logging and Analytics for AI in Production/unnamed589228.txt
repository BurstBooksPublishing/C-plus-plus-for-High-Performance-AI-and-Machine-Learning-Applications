#include <cmath>
#include <cstdlib>
#include <iostream>
#include <numeric>
#include <stdexcept>
#include <vector>

// Compute Mean Squared Error; throws if sizes differ
double compute_mse(const std::vector<double>& actual,
                   const std::vector<double>& predicted)
{
    if (actual.size() != predicted.size())
        throw std::invalid_argument("Size mismatch");

    double sum = std::inner_product(actual.begin(), actual.end(),
                                    predicted.begin(), 0.0,
                                    std::plus<>(),
                                    [](double a, double p) {
                                        double d = a - p;
                                        return d * d;
                                    });
    return sum / static_cast<double>(actual.size());
}

int main()
{
    const std::vector<double> actual    = {1.0, 2.0, 3.0};
    const std::vector<double> predicted = {1.1, 2.1, 3.1};

    try {
        const double mse = compute_mse(actual, predicted);
        std::cout << "Mean Squared Error: " << mse << '\n';
    } catch (const std::exception& ex) {
        std::cerr << "Error: " << ex.what() << '\n';
        return EXIT_FAILURE;
    }
    return EXIT_SUCCESS;
}
