#include <spdlog/spdlog.h>
#include <spdlog/sinks/basic_file_sink.h>
#include <spdlog/async.h>
#include <memory>

int main() {
    // initialise thread-pool for async logging
    spdlog::init_thread_pool(8192, 1);

    // create async file logger with automatic flush on info-level messages
    auto logger = spdlog::basic_logger_mt<spdlog::async_factory>(
        "async_logger", "logs/ai_model.log");
    logger->set_level(spdlog::level::info);
    logger->flush_on(spdlog::level::info);

    // register as default logger so spdlog::... calls use it
    spdlog::set_default_logger(logger);

    spdlog::info("Asynchronous logging enabled.");

    // model inference code would go here

    // ensure all messages are written before exit
    spdlog::shutdown();
    return 0;
}
