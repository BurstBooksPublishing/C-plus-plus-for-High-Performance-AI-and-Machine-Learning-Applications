#include <immintrin.h>   // AVX2 intrinsics
#include <cstddef>       // std::size_t
#include <stdexcept>     // std::runtime_error
#include <cmath>         // std::isfinite

// Add two 32-byte (8×float) AVX2 vectors; srcA and srcB must be 32-byte aligned
void addFloat8(const float* srcA, const float* srcB, float* dst) {
    if (!srcA || !srcB || !dst)
        throw std::runtime_error("Null pointer passed to addFloat8");
    if ((reinterpret_cast<std::uintptr_t>(srcA) & 31) ||
        (reinterpret_cast<std::uintptr_t>(srcB) & 31) ||
        (reinterpret_cast<std::uintptr_t>(dst) & 31))
        throw std::runtime_error("Misaligned pointer passed to addFloat8");

    __m256 a = _mm256_load_ps(srcA);
    __m256 b = _mm256_load_ps(srcB);
    __m256 r = _mm256_add_ps(a, b);
    _mm256_store_ps(dst, r);
}
