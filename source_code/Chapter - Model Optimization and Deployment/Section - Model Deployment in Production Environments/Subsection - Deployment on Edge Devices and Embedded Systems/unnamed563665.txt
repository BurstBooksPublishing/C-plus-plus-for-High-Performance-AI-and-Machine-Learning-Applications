#include <cstdint>
#include "stm32f4xx.h"          // Device-specific header from CMSIS

extern "C" void SystemInit();   // Forward declaration for startup code

// Weak alias allows user override; actual SystemInit() is provided by vendor.
extern "C" void DefaultSystemInit() __attribute__((weak, alias("SystemInit")));

int main()
{
    HAL_Init();                 // Configure Flash prefetch, SysTick, NVIC grouping
    SystemClock_Config();       // Ensure CPU runs at the desired frequency

    while (true) {
        __WFI();                // Sleep until next interrupt (low-power inference)
    }
}

// Minimal PLL setup for STM32F4 @ 168 MHz (HSE 8 MHz)
extern "C" void SystemClock_Config()
{
    RCC->CR |= RCC_CR_HSEON;                           // Enable HSE
    while (!(RCC->CR & RCC_CR_HSERDY));

    RCC->APB1ENR |= RCC_APB1ENR_PWREN;
    PWR->CR |= PWR_CR_VOS;                             // Scale 1 for 168 MHz

    RCC->CFGR |= RCC_CFGR_HPRE_DIV1                    // AHB  = 168 MHz
               | RCC_CFGR_PPRE1_DIV4                   // APB1 =  42 MHz
               | RCC_CFGR_PPRE2_DIV2;                  // APB2 =  84 MHz

    RCC->PLLCFGR = (8UL << RCC_PLLCFGR_PLLM_Pos)       // VCO input  = 1 MHz
                 | (336UL << RCC_PLLCFGR_PLLN_Pos)     // VCO output = 336 MHz
                 | (0UL << RCC_PLLCFGR_PLLP_Pos)       // SYSCLK     = 168 MHz
                 | RCC_PLLCFGR_PLLSRC_HSE
                 | (7UL << RCC_PLLCFGR_PLLQ_Pos);      // USB/SDIO   = 48 MHz

    RCC->CR |= RCC_CR_PLLON;
    while (!(RCC->CR & RCC_CR_PLLRDY));

    FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN | FLASH_ACR_DCEN | FLASH_ACR_LATENCY_5WS;

    RCC->CFGR |= RCC_CFGR_SW_PLL;
    while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL);
}
