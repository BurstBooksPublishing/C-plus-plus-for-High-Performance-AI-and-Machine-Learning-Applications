#include <vector>
#include <cmath>
#include <stdexcept>
#include <limits>

std::vector<uint8_t> quantize(const std::vector<float>& data, float min, float max) {
    if (min >= max) throw std::invalid_argument("min must be less than max");
    if (data.empty()) return {};

    const float scale = 255.0f / (max - min);
    std::vector<uint8_t> quantized;
    quantized.reserve(data.size());

    for (float v : data) {
        float clamped = std::fmax(min, std::fmin(max, v));
        quantized.push_back(static_cast<uint8_t>(std::round((clamped - min) * scale)));
    }
    return quantized;
}

std::vector<float> dequantize(const std::vector<uint8_t>& quantized, float min, float max) {
    if (min >= max) throw std::invalid_argument("min must be less than max");
    if (quantized.empty()) return {};

    const float scale = (max - min) / 255.0f;
    std::vector<float> data;
    data.reserve(quantized.size());

    for (uint8_t q : quantized) {
        data.push_back(q * scale + min);
    }
    return data;
}
