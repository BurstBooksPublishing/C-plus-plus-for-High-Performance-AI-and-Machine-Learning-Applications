#include <vector>
#include <immintrin.h>
#include <stdexcept>

void simd_inference(const std::vector<float>& quantized_weights,
                    const std::vector<float>& pruned_weights,
                    const std::vector<float>& input,
                    std::vector<float>& output)
{
    if (quantized_weights.size() != pruned_weights.size())
        throw std::invalid_argument("Weight vectors must have equal size");
    if (input.size() % 8 != 0)
        throw std::invalid_argument("Input size must be a multiple of 8");
    output.resize(input.size());

    const size_t blocks = input.size() / 8;
    for (size_t b = 0; b < blocks; ++b) {
        __m256 acc = _mm256_setzero_ps();
        const __m256 in = _mm256_loadu_ps(&input[b * 8]);
        for (size_t j = 0; j < quantized_weights.size(); ++j) {
            const __m256 w = _mm256_set1_ps(quantized_weights[j] * pruned_weights[j]);
            acc = _mm256_fmadd_ps(w, in, acc);        // fused multiply-add
        }
        _mm256_storeu_ps(&output[b * 8], acc);
    }
}
