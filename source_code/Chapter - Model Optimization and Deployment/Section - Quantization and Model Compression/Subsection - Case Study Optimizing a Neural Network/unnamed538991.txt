#include <vector>
#include <cmath>
#include <cstdint>
#include <limits>

std::vector<uint8_t> quantize(const std::vector<float>& weights,
                              float scale,
                              int32_t zero_point)
{
    std::vector<uint8_t> quantized_weights;
    quantized_weights.reserve(weights.size());

    for (float w : weights) {
        int32_t q = static_cast<int32_t>(std::round(w / scale)) + zero_point;
        // Clamp to valid uint8 range
        q = std::max(0, std::min(255, q));
        quantized_weights.push_back(static_cast<uint8_t>(q));
    }
    return quantized_weights;
}
