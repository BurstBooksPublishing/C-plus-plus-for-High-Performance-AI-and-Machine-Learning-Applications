#include <Eigen/Core>
#include <stdexcept>

void iterativePruningAndRetraining(Eigen::MatrixXd& weights,
                                   const Eigen::MatrixXd& data,
                                   const Eigen::VectorXd& labels,
                                   int iterations,
                                   double threshold)
{
    if (iterations <= 0)              throw std::invalid_argument("iterations must be positive");
    if (threshold < 0)                throw std::invalid_argument("threshold must be non-negative");
    if (weights.rows() != data.cols())throw std::invalid_argument("weights cols must match data cols");

    for (int i = 0; i < iterations; ++i) {
        pruneWeights(weights, threshold);   // in-place sparsification
        trainModel(weights, data, labels);  // in-place weight update
    }
}
