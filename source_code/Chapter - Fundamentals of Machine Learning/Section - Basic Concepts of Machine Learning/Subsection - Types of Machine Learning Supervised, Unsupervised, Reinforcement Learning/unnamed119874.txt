#include <algorithm>
#include <chrono>
#include <cmath>
#include <iostream>
#include <random>
#include <vector>

constexpr int num_states = 6;
constexpr int num_actions = 2;
constexpr double alpha = 0.1;   // Learning rate
constexpr double gamma = 0.9;   // Discount factor
constexpr int episodes = 1000;
constexpr double epsilon = 0.1; // Exploration rate

std::vector<std::vector<double>> Q(num_states, std::vector<double>(num_actions, 0.0));

std::mt19937 rng(std::chrono::steady_clock::now().time_since_epoch().count());

int choose_action(int state, double eps) {
    std::uniform_real_distribution<double> dist(0.0, 1.0);
    if (dist(rng) < eps) {
        std::uniform_int_distribution<int> act(0, num_actions - 1);
        return act(rng);
    }
    return static_cast<int>(std::distance(
        Q[state].begin(), std::max_element(Q[state].begin(), Q[state].end())));
}

void update_q(int state, int action, int next_state, double reward) {
    double max_q_next = *std::max_element(Q[next_state].begin(), Q[next_state].end());
    Q[state][action] += alpha * (reward + gamma * max_q_next - Q[state][action]);
}

int main() {
    for (int ep = 0; ep < episodes; ++ep) {
        int state = 0;
        while (state != num_states - 1) {
            int action = choose_action(state, epsilon);
            int next_state = state + action + 1; // deterministic right walk
            double reward = (next_state == num_states - 1) ? 1.0 : 0.0;
            update_q(state, action, next_state, reward);
            state = next_state;
        }
    }
}
