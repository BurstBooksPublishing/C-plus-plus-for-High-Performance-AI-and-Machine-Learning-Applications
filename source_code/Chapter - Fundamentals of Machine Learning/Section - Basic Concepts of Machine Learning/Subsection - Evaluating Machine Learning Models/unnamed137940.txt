#include <vector>
#include <stdexcept>

std::vector<std::vector<int>>
confusion_matrix(const std::vector<int>& true_labels,
                 const std::vector<int>& predicted_labels,
                 int num_classes)
{
    if (true_labels.size() != predicted_labels.size())
        throw std::invalid_argument("Label vectors must be the same length.");
    if (num_classes <= 0)
        throw std::invalid_argument("num_classes must be positive.");

    std::vector<std::vector<int>> matrix(num_classes,
                                         std::vector<int>(num_classes, 0));

    for (size_t i = 0; i < true_labels.size(); ++i) {
        int t = true_labels[i];
        int p = predicted_labels[i];
        if (t < 0 || t >= num_classes || p < 0 || p >= num_classes)
            throw std::out_of_range("Label value out of range.");
        ++matrix[t][p];
    }
    return matrix;
}
