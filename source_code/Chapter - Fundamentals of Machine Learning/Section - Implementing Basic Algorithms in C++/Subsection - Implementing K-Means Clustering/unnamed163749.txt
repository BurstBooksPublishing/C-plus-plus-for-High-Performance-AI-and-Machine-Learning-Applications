#include <cmath>
#include <limits>
#include <random>
#include <vector>

struct Point {
    double x{}, y{};
};

double euclideanDistance(const Point& a, const Point& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return std::sqrt(dx * dx + dy * dy);
}

std::vector<Point> kMeansClustering(const std::vector<Point>& data,
                                    int k,
                                    int maxIterations) {
    if (k <= 0 || data.size() < static_cast<size_t>(k) || maxIterations <= 0)
        return {};

    std::vector<Point> centroids(k);
    std::vector<int> assignments(data.size());

    // Seed centroids with distinct random points
    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> dis(0, static_cast<int>(data.size() - 1));
    for (int i = 0; i < k; ++i) {
        centroids[i] = data[dis(gen)];
    }

    for (int iter = 0; iter < maxIterations; ++iter) {
        // Assignment step
        for (size_t i = 0; i < data.size(); ++i) {
            double minDist = std::numeric_limits<double>::max();
            int best = 0;
            for (int j = 0; j < k; ++j) {
                const double d = euclideanDistance(data[i], centroids[j]);
                if (d < minDist) {
                    minDist = d;
                    best = j;
                }
            }
            assignments[i] = best;
        }

        // Update step
        std::vector<Point> sum(k, {0.0, 0.0});
        std::vector<size_t> count(k, 0);
        for (size_t i = 0; i < data.size(); ++i) {
            const int c = assignments[i];
            sum[c].x += data[i].x;
            sum[c].y += data[i].y;
            ++count[c];
        }
        for (int j = 0; j < k; ++j) {
            if (count[j]) {
                centroids[j].x = sum[j].x / count[j];
                centroids[j].y = sum[j].y / count[j];
            }
        }
    }
    return centroids;
}
