#include <memory>
#include <iostream>

#include "tensorflow/core/framework/tensor.h"
#include "tensorflow/core/public/session.h"
#include "tensorflow/core/platform/env.h"
#include "tensorflow/core/platform/status.h"

int main() {
    // Build and start a TensorFlow session.
    std::unique_ptr<tensorflow::Session> session(tensorflow::NewSession(tensorflow::SessionOptions()));
    if (!session) {
        std::cerr << "Failed to create session\n";
        return 1;
    }

    // Create a 2×2 float tensor and fill it.
    tensorflow::Tensor tensor(tensorflow::DT_FLOAT, tensorflow::TensorShape({2, 2}));
    auto tensor_map = tensor.matrix<float>();
    tensor_map(0, 0) = 1.0f;
    tensor_map(0, 1) = 2.0f;
    tensor_map(1, 0) = 3.0f;
    tensor_map(1, 1) = 4.0f;

    // Clean shutdown.
    const tensorflow::Status s = session->Close();
    if (!s.ok()) {
        std::cerr << "Error closing session: " << s.ToString() << "\n";
        return 1;
    }
    return 0;
}
