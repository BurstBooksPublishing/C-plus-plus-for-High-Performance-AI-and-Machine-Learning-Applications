#include <opencv2/opencv.hpp>
#include <iostream>

int main(int argc, char* argv[]) {
    const std::string path = (argc > 1) ? argv[1] : "example.jpg";
    cv::Mat image = cv::imread(path, cv::IMREAD_COLOR);
    if (image.empty()) {
        std::cerr << "Error: Cannot load image from " << path << std::endl;
        return EXIT_FAILURE;
    }

    cv::Mat gray;
    cv::cvtColor(image, gray, cv::COLOR_BGR2GRAY);

    cv::Mat edges;
    cv::Canny(gray, edges, 100, 200);

    cv::imshow("Edges", edges);
    cv::waitKey(0);

    return EXIT_SUCCESS;
}
