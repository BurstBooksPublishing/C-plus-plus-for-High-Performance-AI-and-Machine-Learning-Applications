#include <opencv2/opencv.hpp>
#include <iostream>

int main(int argc, char* argv[]) {
    const std::string path = (argc > 1) ? argv[1] : "example.jpg";

    cv::Mat image = cv::imread(path, cv::IMREAD_COLOR);
    if (image.empty()) {
        std::cerr << "Error: cannot load " << path << std::endl;
        return EXIT_FAILURE;
    }

    cv::Mat resized;
    cv::resize(image, resized, cv::Size(300, 300), 0, 0, cv::INTER_LINEAR);

    cv::imshow("Resized Image", resized);
    cv::waitKey(0);

    return EXIT_SUCCESS;
}
