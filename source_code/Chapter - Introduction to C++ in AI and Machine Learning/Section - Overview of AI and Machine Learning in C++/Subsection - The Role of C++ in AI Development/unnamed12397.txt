#include <iostream>
#include <thread>
#include <vector>
#include <atomic>
#include <chrono>

void task(int id, std::atomic<int>& done) {
    std::cout << "Task " << id << " running on thread " << std::this_thread::get_id() << '\n';
    std::this_thread::sleep_for(std::chrono::milliseconds(100)); // simulate work
    ++done;
}

int main() {
    constexpr int NUM_TASKS = 5;
    std::vector<std::thread> threads;
    threads.reserve(NUM_TASKS);

    std::atomic<int> done{0};

    for (int i = 0; i < NUM_TASKS; ++i) {
        threads.emplace_back(task, i, std::ref(done));
    }

    for (auto& t : threads) {
        t.join();
    }

    std::cout << "All tasks finished. Done count: " << done.load() << '\n';
    return 0;
}
