#include <vector>
#include <iostream>

int main() {
    const std::vector<int> data{1, 2, 3, 4, 5};   // immutable input data

    for (int value : data)                        // range-based loop for clarity
        std::cout << value << ' ';

    std::cout << std::endl;                       // flush and newline
    return 0;
}
