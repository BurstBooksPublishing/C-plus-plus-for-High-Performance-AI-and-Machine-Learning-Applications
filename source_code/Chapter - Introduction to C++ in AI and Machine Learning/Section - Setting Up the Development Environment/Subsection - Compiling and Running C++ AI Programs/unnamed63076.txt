#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <stdexcept>

// Detected language: C++17
// Minimal, production-ready example: fast Mersenne-Twister PRNG + vector sort

int main() {
    constexpr std::size_t kSize = 1'000'000;          // problem size
    std::vector<int> data(kSize);

    // seed with high-resolution clock
    auto seed = static_cast<std::uint_fast64_t>(
        std::chrono::high_resolution_clock::now().time_since_epoch().count());
    std::mt19937_64 rng(seed);                        // 64-bit Mersenne Twister
    std::uniform_int_distribution<int> dist(0, 1'000'000);

    // fill vector
    std::generate(data.begin(), data.end(), [&] { return dist(rng); });

    // sort
    std::sort(data.begin(), data.end());

    // basic sanity check
    if (!std::is_sorted(data.begin(), data.end()))
        throw std::runtime_error("sort failed");

    std::cout << "smallest: " << data.front() << '\n';
    std::cout << "largest:  " << data.back()  << '\n';
    return 0;
}
