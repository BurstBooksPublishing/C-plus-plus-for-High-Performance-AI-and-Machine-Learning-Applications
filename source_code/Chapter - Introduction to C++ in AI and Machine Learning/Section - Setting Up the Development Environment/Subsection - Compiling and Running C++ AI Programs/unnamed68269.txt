#include <iostream>
#include <vector>
#include <string>
#include <stdexcept>
#include <memory>
#include <algorithm>
#include <random>
#include <chrono>

// Example production-ready AI skeleton: a tiny neural-network inference engine.
class Tensor {
public:
    explicit Tensor(std::vector<size_t> shape)
        : shape_(std::move(shape)), size_(computeSize(shape_)) {
        data_.resize(size_, 0.0f);
    }

    float& at(const std::vector<size_t>& idx) {
        return data_[flatten(idx)];
    }
    const float& at(const std::vector<size_t>& idx) const {
        return data_[flatten(idx)];
    }

    size_t size() const { return size_; }
    const std::vector<size_t>& shape() const { return shape_; }

private:
    std::vector<size_t> shape_;
    size_t size_;
    std::vector<float> data_;

    static size_t computeSize(const std::vector<size_t>& s) {
        return std::accumulate(s.begin(), s.end(), 1ULL, std::multiplies<>());
    }
    size_t flatten(const std::vector<size_t>& idx) const {
        size_t pos = 0, stride = 1;
        for (int i = static_cast<int>(shape_.size()) - 1; i >= 0; --i) {
            pos += idx[i] * stride;
            stride *= shape_[i];
        }
        return pos;
    }
};

class Layer {
public:
    virtual ~Layer() = default;
    virtual Tensor forward(const Tensor& in) const = 0;
};

class DenseLayer : public Layer {
public:
    DenseLayer(size_t in, size_t out)
        : weights_({in, out}), biases_({out}) {
        std::mt19937 rng(std::random_device{}());
        std::uniform_real_distribution<float> dist(-0.1f, 0.1f);
        for (size_t i = 0; i < weights_.size(); ++i) {
            weights_.data()[i] = dist(rng);
        }
        for (size_t i = 0; i < biases_.size(); ++i) {
            biases_.data()[i] = dist(rng);
        }
    }

    Tensor forward(const Tensor& in) const override {
        if (in.shape().size() != 2 || in.shape()[1] != weights_.shape()[0])
            throw std::invalid_argument("Input shape mismatch");
        Tensor out({in.shape()[0], weights_.shape()[1]});
        for (size_t b = 0; b < in.shape()[0]; ++b) {
            for (size_t j = 0; j < weights_.shape()[1]; ++j) {
                float sum = biases_.at({j});
                for (size_t i = 0; i < weights_.shape()[0]; ++i) {
                    sum += in.at({b, i}) * weights_.at({i, j});
                }
                out.at({b, j}) = sum;
            }
        }
        return out;
    }

private:
    Tensor weights_;
    Tensor biases_;
};

class ReLU : public Layer {
public:
    Tensor forward(const Tensor& in) const override {
        Tensor out(in.shape());
        for (size_t i = 0; i < in.size(); ++i) {
            out.data()[i] = std::max(0.0f, in.data()[i]);
        }
        return out;
    }
};

class Network {
public:
    void add(std::unique_ptr<Layer> layer) {
        layers_.emplace_back(std::move(layer));
    }

    Tensor predict(const Tensor& in) const {
        Tensor tmp = in;
        for (const auto& layer : layers_) {
            tmp = layer->forward(tmp);
        }
        return tmp;
    }

private:
    std::vector<std::unique_ptr<Layer>> layers_;
};

int main() {
    Network net;
    net.add(std::make_unique<DenseLayer>(4, 8));
    net.add(std::make_unique<ReLU>());
    net.add(std::make_unique<DenseLayer>(8, 3));

    Tensor input({1, 4});
    for (size_t i = 0; i < 4; ++i) input.at({0, i}) = static_cast<float>(i);

    Tensor output = net.predict(input);
    for (size_t i = 0; i < 3; ++i)
        std::cout << output.at({0, i}) << (i + 1 == 3 ? '\n' : ' ');
    return 0;
}
