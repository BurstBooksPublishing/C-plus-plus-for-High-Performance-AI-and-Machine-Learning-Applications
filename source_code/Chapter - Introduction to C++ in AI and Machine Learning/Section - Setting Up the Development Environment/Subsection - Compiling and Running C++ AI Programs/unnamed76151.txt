#include <iostream>
#include <vector>
#include <string>
#include <stdexcept>
#include <algorithm>
#include <random>
#include <chrono>

// Simple linear regression model for demonstration
class LinearModel {
public:
    LinearModel(double w = 0.0, double b = 0.0) : weight(w), bias(b) {}

    double predict(double x) const {
        return weight * x + bias;
    }

    void fit(const std::vector<double>& xs,
             const std::vector<double>& ys,
             size_t epochs,
             double lr) {
        if (xs.size() != ys.size() || xs.empty())
            throw std::invalid_argument("Invalid data");

        const size_t n = xs.size();
        for (size_t e = 0; e < epochs; ++e) {
            double grad_w = 0.0, grad_b = 0.0;
            for (size_t i = 0; i < n; ++i) {
                double err = predict(xs[i]) - ys[i];
                grad_w += err * xs[i];
                grad_b += err;
            }
            weight -= lr * grad_w / n;
            bias   -= lr * grad_b / n;
        }
    }

private:
    double weight, bias;
};

int main() {
    std::vector<double> xs{1, 2, 3, 4, 5};
    std::vector<double> ys{3, 5, 7, 9, 11};  // y = 2x + 1

    LinearModel model;
    model.fit(xs, ys, 10000, 0.01);

    std::cout << "Predict y for x=6: " << model.predict(6) << '\n';
    return 0;
}
