#include <tensorflow/cc/client/client_session.h>
#include <tensorflow/cc/ops/standard_ops.h>
#include <tensorflow/cc/framework/ops.h>
#include <tensorflow/core/framework/tensor.h>
#include <iostream>

int main() {
    using namespace tensorflow;
    using namespace tensorflow::ops;

    Scope root = Scope::NewRootScope();

    // Build a trivial graph: c = a + b
    auto a = Placeholder(root.WithOpName("a"), DT_FLOAT);
    auto b = Placeholder(root.WithOpName("b"), DT_FLOAT);
    auto c = Add(root.WithOpName("c"), a, b);

    // Create session and run
    ClientSession session(root);
    std::vector<Tensor> outputs;
    TF_CHECK_OK(session.Run({{a, {1.0f}}, {b, {2.0f}}}, {c}, &outputs));

    std::cout << outputs[0].scalar<float>()() << std::endl; // 3
    return 0;
}
