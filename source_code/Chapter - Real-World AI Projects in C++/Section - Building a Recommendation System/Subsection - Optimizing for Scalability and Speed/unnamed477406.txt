    int provided;
    MPI_Init_thread(&argc, &argv, MPI_THREAD_FUNNELED, &provided);
    if (provided < MPI_THREAD_FUNNELED) {
        std::cerr << "MPI thread support insufficient\n";
        std::exit(EXIT_FAILURE);
    }

    MPI_Finalize();

}
