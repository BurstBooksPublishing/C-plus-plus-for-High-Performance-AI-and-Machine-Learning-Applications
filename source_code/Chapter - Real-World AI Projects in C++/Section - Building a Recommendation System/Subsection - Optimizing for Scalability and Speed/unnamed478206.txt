#include <sw/redis++/redis.h>
#include <iostream>

int main() {
    try {
        sw::redis::Redis redis("tcp://127.0.0.1:6379");
        redis.set("user_123_similarity", "0.85");
    } catch (const sw::redis::Error &e) {
        std::cerr << "Redis error: " << e.what() << std::endl;
        return 1;
    }
    return 0;
}
