#include <onnxruntime_cxx_api.h>
#include <vector>
#include <stdexcept>

// Assumes: session, create_input_tensor(tokens,mask) are defined elsewhere
std::vector<float> infer_sentiment(const std::vector<int64_t>& tokens,
                                   const std::vector<int64_t>& mask) {
    Ort::MemoryInfo memory_info = Ort::MemoryInfo::CreateCpu(OrtArenaAllocator, OrtMemTypeDefault);

    auto input_tensor = create_input_tensor(tokens, mask);          // shape: {1, seq_len}

    const char* input_names[]  = {"input_ids", "attention_mask"};
    const char* output_names[] = {"output"};

    std::vector<Ort::Value> inputs;
    inputs.reserve(2);
    inputs.push_back(std::move(input_tensor.first));
    inputs.push_back(std::move(input_tensor.second));

    auto outputs = session.Run(Ort::RunOptions{nullptr},
                               input_names,  inputs.data(),  inputs.size(),
                               output_names, 1);

    if (outputs.size() != 1 || !outputs[0].IsTensor())
        throw std::runtime_error("Invalid model output");

    float* output_data = outputs[0].GetTensorMutableData<float>();
    const size_t count = outputs[0].GetTensorTypeAndShapeInfo().GetElementCount();

    return std::vector<float>(output_data, output_data + count);
}
