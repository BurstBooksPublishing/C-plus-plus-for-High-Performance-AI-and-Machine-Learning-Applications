#include <cuda_runtime.h>

__global__ void relu(const float* __restrict__ input,
                     float* __restrict__ output,
                     const int N)
{
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        output[idx] = fmaxf(0.0f, input[idx]);
    }
}
