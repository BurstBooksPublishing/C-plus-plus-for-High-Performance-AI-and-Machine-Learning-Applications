#include <cuda_runtime.h>

constexpr int BLOCK_SIZE = 16;

__global__ void matrixMul(const float* __restrict__ A,
                          const float* __restrict__ B,
                          float* __restrict__ C,
                          int N) {
    __shared__ float sharedA[BLOCK_SIZE][BLOCK_SIZE + 1]; // +1 avoids bank conflicts
    __shared__ float sharedB[BLOCK_SIZE][BLOCK_SIZE + 1];

    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    float sum = 0.0f;

    for (int tile = 0; tile < (N + BLOCK_SIZE - 1) / BLOCK_SIZE; ++tile) {
        int aRow = row;
        int aCol = tile * BLOCK_SIZE + threadIdx.x;
        sharedA[threadIdx.y][threadIdx.x] =
            (aRow < N && aCol < N) ? A[aRow * N + aCol] : 0.0f;

        int bRow = tile * BLOCK_SIZE + threadIdx.y;
        int bCol = col;
        sharedB[threadIdx.y][threadIdx.x] =
            (bRow < N && bCol < N) ? B[bRow * N + bCol] : 0.0f;

        __syncthreads();

        #pragma unroll
        for (int k = 0; k < BLOCK_SIZE; ++k) {
            sum += sharedA[threadIdx.y][k] * sharedB[k][threadIdx.x];
        }
        __syncthreads();
    }

    if (row < N && col < N) C[row * N + col] = sum;
}
