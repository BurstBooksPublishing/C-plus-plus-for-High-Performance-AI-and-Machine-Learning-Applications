#include <cuda_runtime.h>
#include <cstdio>

// Minimal RAII helpers to avoid manual free / reset
struct CudaStream {
    cudaStream_t s{};
    CudaStream() { cudaStreamCreate(&s); }
    ~CudaStream() { if (s) cudaStreamDestroy(s); }
    operator cudaStream_t() const { return s; }
};

struct CudaMem {
    void* p{};
    explicit CudaMem(size_t bytes) { cudaMalloc(&p, bytes); }
    ~CudaMem() { if (p) cudaFree(p); }
    template<typename T> T* as() { return static_cast<T*>(p); }
};

// Host-side wrapper that performs the complete D = A * B pipeline
cudaError_t launchMatMul(const float* h_A,
                         const float* h_B,
                         float*       h_C,
                         int          N,
                         size_t       size)
{
    CudaStream stream;
    CudaMem d_A(size), d_B(size), d_C(size);

    cudaMemcpyAsync(d_A.as<float>(), h_A, size, cudaMemcpyHostToDevice, stream);
    cudaMemcpyAsync(d_B.as<float>(), h_B, size, cudaMemcpyHostToDevice, stream);

    dim3 block(16, 16);
    dim3 grid((N + 15) / 16, (N + 15) / 16);
    matrixMul<<<grid, block, 0, stream>>>(d_A.as<float>(), d_B.as<float>(),
                                          d_C.as<float>(), N);

    cudaMemcpyAsync(h_C, d_C.as<float>(), size, cudaMemcpyDeviceToHost, stream);
    return cudaStreamSynchronize(stream);
}
