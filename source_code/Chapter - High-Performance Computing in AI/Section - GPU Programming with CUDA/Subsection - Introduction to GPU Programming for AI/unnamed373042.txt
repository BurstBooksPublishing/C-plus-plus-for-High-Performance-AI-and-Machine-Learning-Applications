#include <cublas_v2.h>
#include <cuda_runtime.h>

cublasStatus_t launch_gemm(float* d_C,
                           const float* d_A,
                           const float* d_B,
                           int N)
{
    static cublasHandle_t handle = []() {
        cublasHandle_t h;
        cublasCreate(&h);          // one-time creation
        return h;
    }();

    const float alpha = 1.0f;
    const float beta  = 0.0f;

    return cublasSgemm(handle,
                       CUBLAS_OP_N, CUBLAS_OP_N,
                       N, N, N,
                       &alpha,
                       d_A, N,
                       d_B, N,
                       &beta,
                       d_C, N);
}
