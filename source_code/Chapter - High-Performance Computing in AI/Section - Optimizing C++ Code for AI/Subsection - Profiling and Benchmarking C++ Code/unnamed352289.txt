#include <vector>
#include <chrono>
#include <iostream>
#include <stdexcept>
#include <omp.h>

void parallel_matrix_multiply(const std::vector<std::vector<double>>& A,
                              const std::vector<std::vector<double>>& B,
                              std::vector<std::vector<double>>& C) {
    const size_t n = A.size();
    if (B.size() != n || B[0].size() != n || C.size() != n || C[0].size() != n)
        throw std::invalid_argument("Matrices must be square and equally sized");

    #pragma omp parallel for collapse(2) schedule(static)
    for (size_t i = 0; i < n; ++i) {
        for (size_t j = 0; j < n; ++j) {
            double sum = 0.0;
            for (size_t k = 0; k < n; ++k) {
                sum += A[i][k] * B[k][j];
            }
            C[i][j] = sum;
        }
    }
}

int main() {
    constexpr size_t n = 1000;
    std::vector<std::vector<double>> A(n, std::vector<double>(n, 1.0));
    std::vector<std::vector<double>> B(n, std::vector<double>(n, 1.0));
    std::vector<std::vector<double>> C(n, std::vector<double>(n, 0.0));

    const auto start = std::chrono::high_resolution_clock::now();
    parallel_matrix_multiply(A, B, C);
    const auto end = std::chrono::high_resolution_clock::now();

    std::cout << "Parallel multiplication took "
              << std::chrono::duration<double>(end - start).count()
              << " seconds.\n";
    return 0;
}
