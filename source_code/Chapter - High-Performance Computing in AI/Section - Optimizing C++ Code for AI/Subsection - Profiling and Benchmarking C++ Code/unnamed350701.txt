#include <vector>
#include <chrono>
#include <iostream>
#include <stdexcept>

using Matrix = std::vector<std::vector<double>>;

void matrixMultiply(const Matrix& A, const Matrix& B, Matrix& C) {
    const size_t n = A.size();
    if (n == 0 || A[0].size() != n || B.size() != n || B[0].size() != n || C.size() != n || C[0].size() != n)
        throw std::invalid_argument("Matrices must be square and equally sized");

    for (size_t i = 0; i < n; ++i)
        for (size_t j = 0; j < n; ++j) {
            double sum = 0.0;
            for (size_t k = 0; k < n; ++k)
                sum += A[i][k] * B[k][j];
            C[i][j] = sum;
        }
}

int main() {
    constexpr size_t n = 100;
    Matrix A(n, std::vector<double>(n, 1.0));
    Matrix B(n, std::vector<double>(n, 1.0));
    Matrix C(n, std::vector<double>(n, 0.0));

    const auto start = std::chrono::high_resolution_clock::now();
    matrixMultiply(A, B, C);
    const auto end = std::chrono::high_resolution_clock::now();

    std::cout << "Matrix multiplication took "
              << std::chrono::duration<double>(end - start).count()
              << " seconds\n";
    return 0;
}
