#include <immintrin.h>
#include <cstddef>

void elementwise_multiply_avx(const float* a, const float* b, float* result, size_t size) {
    size_t i = 0;
    for (; i + 8 <= size; i += 8) {
        __m256 va = _mm256_loadu_ps(&a[i]);
        __m256 vb = _mm256_loadu_ps(&b[i]);
        __m256 vresult = _mm256_mul_ps(va, vb);
        _mm256_storeu_ps(&result[i], vresult);
    }
    // handle remaining elements
    for (; i < size; ++i) {
        result[i] = a[i] * b[i];
    }
}
