#include <mpi.h>

int main(int argc, char* argv[]) {
    MPI_Init(&argc, &argv);

    int rank;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    int data = 0;               // ensure all ranks have a defined value
    if (rank == 0) data = 42;   // root sets the broadcast value

    MPI_Bcast(&data, 1, MPI_INT, 0, MPI_COMM_WORLD);

    MPI_Finalize();
    return 0;
}
