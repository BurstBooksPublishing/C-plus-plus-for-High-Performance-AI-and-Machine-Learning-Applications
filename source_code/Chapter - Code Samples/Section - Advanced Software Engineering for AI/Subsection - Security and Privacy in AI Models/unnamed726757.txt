#include <torch/torch.h>
#include <iostream>

struct NetImpl : torch::nn::Module {
    NetImpl() :
        fc1(register_module("fc1", torch::nn::Linear(784, 128))),
        fc2(register_module("fc2", torch::nn::Linear(128, 64))),
        fc3(register_module("fc3", torch::nn::Linear(64, 10))) {}

    torch::Tensor forward(torch::Tensor x) {
        x = torch::relu(fc1->forward(x));
        x = torch::relu(fc2->forward(x));
        return torch::log_softmax(fc3->forward(x), 1);
    }

private:
    torch::nn::Linear fc1{nullptr}, fc2{nullptr}, fc3{nullptr};
};
TORCH_MODULE(Net);

int main() {
    torch::manual_seed(1);
    auto device = torch::cuda::is_available() ? torch::kCUDA : torch::kCPU;

    Net net;
    net->to(device);

    auto train_set = torch::data::datasets::MNIST(
                         "./data",
                         torch::data::datasets::MNIST::Mode::kTrain)
                         .map(torch::data::transforms::Normalize<>(0.1307, 0.3081))
                         .map(torch::data::transforms::Stack<>());
    auto train_loader = torch::data::make_data_loader(
                              std::move(train_set),
                              torch::data::DataLoaderOptions().batch_size(64).workers(2));

    torch::optim::SGD optimizer(net->parameters(), torch::optim::SGDOptions(0.01).momentum(0.5));

    net->train();
    for (size_t epoch = 1; epoch <= 10; ++epoch) {
        size_t batch_idx = 0;
        for (auto& batch : *train_loader) {
            auto data = batch.data.view({-1, 784}).to(device);
            auto targets = batch.target.to(device);

            optimizer.zero_grad();
            auto output = net->forward(data);
            auto loss = torch::nll_loss(output, targets);
            loss.backward();
            optimizer.step();

            if (++batch_idx % 100 == 0) {
                std::cout << "Epoch: " << epoch
                          << " | Batch: " << batch_idx
                          << " | Loss: " << loss.item<float>() << std::endl;
            }
        }
    }
    return 0;
}
