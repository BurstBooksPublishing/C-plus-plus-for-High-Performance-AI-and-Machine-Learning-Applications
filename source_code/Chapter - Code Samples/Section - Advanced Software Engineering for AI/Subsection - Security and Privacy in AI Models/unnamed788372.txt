#include <algorithm>
#include <cctype>
#include <iostream>
#include <string>
#include <vector>

// Convert text to lowercase
std::string toLowercase(const std::string& text) {
    std::string lowerText = text;
    std::transform(lowerText.begin(), lowerText.end(), lowerText.begin(),
                  [](unsigned char c) { return std::tolower(c); });
    return lowerText;
}

// Remove punctuation; keep only letters and spaces
std::string removePunctuation(const std::string& text) {
    std::string cleaned;
    cleaned.reserve(text.size());
    for (unsigned char ch : text) {
        if (std::isalpha(ch) || std::isspace(ch)) cleaned.push_back(ch);
    }
    return cleaned;
}

// Split text into whitespace-separated words
std::vector<std::string> tokenize(const std::string& text) {
    std::vector<std::string> tokens;
    std::string word;
    for (unsigned char ch : text) {
        if (std::isspace(ch)) {
            if (!word.empty()) {
                tokens.emplace_back(std::move(word));
                word.clear();
            }
        } else {
            word.push_back(ch);
        }
    }
    if (!word.empty()) tokens.emplace_back(std::move(word));
    return tokens;
}

// Full preprocessing pipeline: lowercase → remove punctuation → tokenize
std::vector<std::string> preprocessText(const std::string& text) {
    return tokenize(removePunctuation(toLowercase(text)));
}

int main() {
    const std::string inputText = "Hello, World! This is a test for NLP preprocessing.";
    const auto processedTokens = preprocessText(inputText);

    for (const auto& token : processedTokens) std::cout << token << '\n';
}
