#include <vector>
#include <iostream>
#include <random>
#include <cmath>
#include <mpi.h>

struct Layer {
    std::vector<float> weights;
    std::vector<float> biases;
    int input_size;
    int output_size;
};

void initialize_layer(Layer& layer, int input_size, int output_size, std::mt19937& gen) {
    layer.input_size = input_size;
    layer.output_size = output_size;
    layer.weights.resize(input_size * output_size);
    layer.biases.resize(output_size);

    std::uniform_real_distribution<float> dist(-0.5f, 0.5f);
    for (auto& w : layer.weights) w = dist(gen);
    for (auto& b : layer.biases) b = dist(gen);
}

std::vector<float> forward_propagate(const Layer& layer, const std::vector<float>& input) {
    std::vector<float> output(layer.output_size, 0.0f);

    for (int i = 0; i < layer.output_size; ++i) {
        float sum = layer.biases[i];
        for (int j = 0; j < layer.input_size; ++j)
            sum += input[j] * layer.weights[j * layer.output_size + i];
        output[i] = 1.0f / (1.0f + std::exp(-sum)); // Sigmoid
    }
    return output;
}

int main(int argc, char** argv) {
    MPI_Init(&argc, &argv);

    int world_rank;
    MPI_Comm_rank(MPI_COMM_WORLD, &world_rank);

    std::mt19937 gen(42 + world_rank); // Deterministic per rank

    Layer layer1, layer2;
    initialize_layer(layer1, 784, 128, gen);
    initialize_layer(layer2, 128, 10, gen);

    std::vector<float> input(784, 0.5f);

    std::vector<float> hidden = forward_propagate(layer1, input);
    std::vector<float> output = forward_propagate(layer2, hidden);

    if (world_rank == 0) {
        std::cout << "Final Output:";
        for (float v : output) std::cout << ' ' << v;
        std::cout << '\n';
    }

    MPI_Finalize();
    return 0;
}
