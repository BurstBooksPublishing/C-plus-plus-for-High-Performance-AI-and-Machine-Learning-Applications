#include <vector>
#include <iostream>
#include <mpi.h>

void distributed_matrix_vector_mult(const std::vector<double>& matrix,
                                    const std::vector<double>& vector,
                                    std::vector<double>& result,
                                    int rows, int cols, int rank, int size) {
    int local_rows = rows / size;
    std::vector<double> local_matrix(local_rows * cols);
    std::vector<double> local_result(local_rows);

    MPI_Scatter(matrix.data(), local_rows * cols, MPI_DOUBLE,
                local_matrix.data(), local_rows * cols, MPI_DOUBLE,
                0, MPI_COMM_WORLD);

    // Broadcast vector to all ranks
    MPI_Bcast(const_cast<double*>(vector.data()), cols, MPI_DOUBLE, 0, MPI_COMM_WORLD);

    for (int i = 0; i < local_rows; ++i) {
        local_result[i] = 0.0;
        for (int j = 0; j < cols; ++j) {
            local_result[i] += local_matrix[i * cols + j] * vector[j];
        }
    }

    MPI_Gather(local_result.data(), local_rows, MPI_DOUBLE,
               result.data(), local_rows, MPI_DOUBLE,
               0, MPI_COMM_WORLD);
}

int main(int argc, char** argv) {
    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    const int rows = 8;
    const int cols = 8;
    std::vector<double> matrix(rows * cols);
    std::vector<double> vector(cols);
    std::vector<double> result(rows);

    if (rank == 0) {
        for (int i = 0; i < rows; ++i)
            for (int j = 0; j < cols; ++j)
                matrix[i * cols + j] = i + j;
        for (int j = 0; j < cols; ++j)
            vector[j] = j;
    }

    distributed_matrix_vector_mult(matrix, vector, result, rows, cols, rank, size);

    if (rank == 0) {
        std::cout << "Result: ";
        for (double val : result) std::cout << val << ' ';
        std::cout << std::endl;
    }

    MPI_Finalize();
    return 0;
}
