#include <torch/torch.h>
#include <torchvision/models/resnet.h>

struct CustomModelImpl : torch::nn::Module {
    CustomModelImpl(int64_t num_classes = 10) {
        // load pretrained weights and replace the classifier
        resnet = torch::vision::models::resnet18(
            torch::vision::models::ResNet18_Weights::DEFAULT);
        resnet->fc = torch::nn::Linear(resnet->fc->options.in_features(), num_classes);
        register_module("resnet", resnet);
    }

    torch::Tensor forward(torch::Tensor x) {
        return resnet->forward(x);
    }

    torch::nn::Sequential resnet;
};
TORCH_MODULE(CustomModel);

int main() {
    torch::manual_seed(0);
    torch::Device device(torch::cuda::is_available() ? torch::kCUDA : torch::kCPU);

    CustomModel model;
    model->to(device);
    model->eval();                       // inference mode

    torch::NoGradGuard no_grad;          // disable autograd for speed

    auto input = torch::randn({1, 3, 224, 224}, device);
    auto output = model->forward(input);

    std::cout << output << std::endl;
    return 0;
}
