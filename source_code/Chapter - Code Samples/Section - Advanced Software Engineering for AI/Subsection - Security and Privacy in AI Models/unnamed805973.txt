#include <torch/script.h>
#include <torch/torch.h>
#include <vector>
#include <iostream>
#include <stdexcept>

torch::jit::script::Module load_model(const std::string& model_path) {
    torch::jit::script::Module model = torch::jit::load(model_path);
    model.eval();                       // disable dropout/batch-norm updates
    return model;
}

torch::Tensor preprocess_input(const std::vector<float>& input_data) {
    return torch::from_blob(const_cast<float*>(input_data.data()),
                            {1, static_cast<int64_t>(input_data.size())},
                            torch::kFloat32)
           .clone();                    // ensure ownership
}

std::vector<float> run_inference(torch::jit::script::Module& model,
                                 const torch::Tensor& input_tensor) {
    torch::NoGradGuard no_grad;         // turn off autograd for speed
    torch::Tensor output = model.forward({input_tensor}).toTensor();
    return std::vector<float>(output.data_ptr<float>(),
                              output.data_ptr<float>() + output.numel());
}

int main() {
    const std::string model_path = "model.pt";
    torch::jit::script::Module model = load_model(model_path);

    const std::vector<float> input_data{0.1f, 0.2f, 0.3f, 0.4f, 0.5f};
    torch::Tensor input_tensor = preprocess_input(input_data);

    std::vector<float> output_data = run_inference(model, input_tensor);

    for (float v : output_data) std::cout << v << ' ';
    std::cout << '\n';
    return 0;
}
