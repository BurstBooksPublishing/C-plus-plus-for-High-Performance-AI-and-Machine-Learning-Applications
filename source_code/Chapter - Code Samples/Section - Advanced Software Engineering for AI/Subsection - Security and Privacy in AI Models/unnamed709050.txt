#include <iostream>
#include <vector>
#include <random>
#include <stdexcept>
#include <cmath>

class NeuralNetwork {
public:
    NeuralNetwork(std::size_t input_size, std::size_t output_size)
        : weights_(input_size, std::vector<double>(output_size)),
          biases_(output_size),
          activations_(output_size) {
        if (input_size == 0 || output_size == 0)
            throw std::invalid_argument("Layer sizes must be > 0");

        std::mt19937 rng(std::random_device{}());
        std::uniform_real_distribution<double> dist(-1.0, 1.0);

        for (auto& row : weights_)
            for (auto& w : row) w = dist(rng);
        for (auto& b : biases_) b = dist(rng);
    }

    std::vector<double> forward(const std::vector<double>& input) const {
        if (input.size() != weights_.size())
            throw std::invalid_argument("Input size mismatch");

        std::vector<double> output(biases_.size(), 0.0);
        for (std::size_t j = 0; j < biases_.size(); ++j) {
            double sum = biases_[j];
            for (std::size_t i = 0; i < input.size(); ++i)
                sum += input[i] * weights_[i][j];
            output[j] = sigmoid(sum);
        }
        return output;
    }

private:
    static double sigmoid(double x) { return 1.0 / (1.0 + std::exp(-x)); }

    std::vector<std::vector<double>> weights_;
    std::vector<double> biases_;
    mutable std::vector<double> activations_; // not used in this version
};

int main() {
    NeuralNetwork nn(3, 2);
    std::vector<double> input{1.0, 0.5, -1.0};
    auto output = nn.forward(input);

    for (double v : output) std::cout << v << ' ';
    std::cout << '\n';
}
