#include #include #include class LinearRegression { private: std::vector weights; double bias; double learning_rate; int epochs; public: LinearRegression(double lr, int e) : learning_rate(lr), epochs(e) {} void train(const std::vector>& X, const std::vector& y) { int n_features = X[0].size(); weights.resize(n_features, 0.0); bias = 0.0; for (int epoch = 0; epoch < epochs; ++epoch) { double total_loss = 0.0; std::vector grad_weights(n_features, 0.0); double grad_bias = 0.0; for (size_t i = 0; i < X.size(); ++i) { double prediction = predict(X[i]); double error = prediction - y[i]; total_loss += error * error; for (int j = 0; j < n_features; ++j) { grad_weights[j] += error * X[i][j]; } grad_bias += error; } for (int j = 0; j < n_features; ++j) { weights[j] -= (learning_rate / X.size()) * grad_weights[j]; } bias -= (learning_rate / X.size()) * grad_bias; if (epoch % 100 == 0) { std::cout << "Epoch " << epoch << ", Loss: " << total_loss / X.size() << std::endl; } } } double predict(const std::vector& x) { double result = bias; for (size_t i = 0; i < x.size(); ++i) { result += weights[i] * x[i]; } return result; } void evaluate(const std::vector>& X, const std::vector& y) { double total_error = 0.0; for (size_t i = 0; i < X.size(); ++i) { double prediction = predict(X[i]); double error = prediction - y[i]; total_error += error * error; } std::cout << "Mean Squared Error: " << total_error / X.size() << std::endl; } }; int main() { std::vector> X = {{1, 2}, {2, 3}, {3, 4}, {4, 5}}; std::vector y = {3, 5, 7, 9}; LinearRegression model(0.01, 1000); model.train(X, y); model.evaluate(X, y); return 0; }
