#include <fstream>
#include <vector>

bool writeFloatVector(const std::vector<float>& data, const std::string& path) {
    std::ofstream out(path, std::ios::binary);
    if (!out) return false;                       // open failed
    out.write(reinterpret_cast<const char*>(data.data()),
              data.size() * sizeof(float));
    return out.good();                            // true only if all writes succeeded
}
