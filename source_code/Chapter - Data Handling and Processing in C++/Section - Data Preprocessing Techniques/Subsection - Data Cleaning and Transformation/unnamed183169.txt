#include <vector>
#include <numeric>
#include <cmath>
#include <stdexcept>

std::vector<double> standardize(const std::vector<double>& data)
{
    if (data.empty()) throw std::invalid_argument("empty data");

    const double mean = std::accumulate(data.begin(), data.end(), 0.0) / data.size();

    double variance = 0.0;
    for (double v : data) variance += (v - mean) * (v - mean);
    variance /= data.size();

    const double std_dev = std::sqrt(variance);
    if (std_dev == 0.0) throw std::invalid_argument("zero variance");

    std::vector<double> out(data.size());
    for (size_t i = 0; i < data.size(); ++i) out[i] = (data[i] - mean) / std_dev;
    return out;
}
