#include <numeric>
#include <cmath>
#include <vector>
#include <stdexcept>

void standardize(std::vector<double>& data) {
    if (data.empty()) return;                       // guard against division by zero
    const size_t n = data.size();

    const double mean = std::accumulate(data.begin(), data.end(), 0.0) / n;

    double accum = 0.0;
    for (double v : data) accum += (v - mean) * (v - mean);
    const double stddev = std::sqrt(accum / n);

    if (stddev == 0.0) return;                      // all values identical; leave unchanged
    for (double& v : data) v = (v - mean) / stddev;
}
