#include <algorithm>
#include <stdexcept>
#include <vector>

void normalize(std::vector<double>& data) {
    if (data.empty()) return;                       // nothing to do

    const auto [min_it, max_it] = std::minmax_element(data.begin(), data.end());
    const double span = *max_it - *min_it;
    if (span == 0.0) {                              // all values identical
        std::fill(data.begin(), data.end(), 0.0);
        return;
    }

    for (double& v : data) v = (v - *min_it) / span;
}
