#include <fstream>
#include <vector>

bool writeFloatVector(const std::vector<float>& data, const std::string& path) {
    std::ofstream file(path, std::ios::binary);
    if (!file) return false;                       // open failed
    file.write(reinterpret_cast<const char*>(data.data()),
               data.size() * sizeof(float));
    return file.good();                            // true only if all writes succeeded
}
