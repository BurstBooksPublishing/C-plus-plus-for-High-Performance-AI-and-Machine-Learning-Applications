#include <vector>
#include <stdexcept>

void update_parameters(std::vector<double>& theta,
                       const std::vector<double>& gradient,
                       double alpha)
{
    if (theta.size() != gradient.size())
        throw std::invalid_argument("theta and gradient must have the same size");

    for (size_t i = 0; i < theta.size(); ++i)
        theta[i] -= alpha * gradient[i];
}
