#include <vector>
#include <numeric>
#include <cmath>
#include <stdexcept>

class DataPreprocessor {
public:
    // Returns Z-score normalized copy; throws if input is empty or std-dev is zero
    std::vector<double> normalize(const std::vector<double>& data) {
        if (data.empty()) throw std::invalid_argument("empty data");
        const double mean = calculate_mean(data);
        const double std_dev = calculate_std_dev(data, mean);
        if (std_dev == 0.0) throw std::invalid_argument("zero variance");
        std::vector<double> normalized_data;
        normalized_data.reserve(data.size());
        for (double value : data) {
            normalized_data.push_back((value - mean) / std_dev);
        }
        return normalized_data;
    }

private:
    double calculate_mean(const std::vector<double>& data) {
        return std::accumulate(data.begin(), data.end(), 0.0) / data.size();
    }

    double calculate_std_dev(const std::vector<double>& data, double mean) {
        double variance = 0.0;
        for (double value : data) {
            variance += std::pow(value - mean, 2);
        }
        return std::sqrt(variance / data.size());
    }
};
