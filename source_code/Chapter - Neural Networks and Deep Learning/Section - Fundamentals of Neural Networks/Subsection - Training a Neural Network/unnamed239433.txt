#include <iostream>
#include <vector>

constexpr int    max_iterations = 1000;
constexpr double learning_rate  = 0.01;

// Minimal training loop: forward → loss → back-prop
void train(const std::vector<double>& input,
           const std::vector<double>& targets,
           std::vector<double>& weights,
           std::vector<double>& biases)
{
    for (int iter = 0; iter < max_iterations; ++iter) {
        std::vector<double> output = forward(input, weights, biases);
        double loss = mse_loss(output, targets);
        backprop(input, output, targets, weights, biases, learning_rate);
        if (iter % 100 == 0)
            std::cout << "Iteration " << iter << ", Loss: " << loss << '\n';
    }
}
