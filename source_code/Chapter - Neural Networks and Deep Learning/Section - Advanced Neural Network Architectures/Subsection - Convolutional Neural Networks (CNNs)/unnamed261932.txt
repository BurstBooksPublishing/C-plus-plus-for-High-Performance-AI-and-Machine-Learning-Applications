#include <vector>
#include <stdexcept>

std::vector<std::vector<float>>
convolve2D(const std::vector<std::vector<float>>& input,
           const std::vector<std::vector<float>>& kernel)
{
    if (input.empty() || kernel.empty()) throw std::invalid_argument("Empty matrix");
    const size_t inH = input.size();
    const size_t inW = input.front().size();
    const size_t kH  = kernel.size();
    const size_t kW  = kernel.front().size();

    if (kH > inH || kW > inW) throw std::invalid_argument("Kernel larger than input");
    for (const auto& row : input)  if (row.size() != inW) throw std::invalid_argument("Irregular input");
    for (const auto& row : kernel) if (row.size() != kW)  throw std::invalid_argument("Irregular kernel");

    const size_t outH = inH - kH + 1;
    const size_t outW = inW - kW + 1;
    std::vector<std::vector<float>> output(outH, std::vector<float>(outW, 0.0f));

    for (size_t i = 0; i < outH; ++i)
        for (size_t j = 0; j < outW; ++j)
            for (size_t m = 0; m < kH; ++m)
                for (size_t n = 0; n < kW; ++n)
                    output[i][j] += input[i + m][j + n] * kernel[m][n];

    return output;
}
