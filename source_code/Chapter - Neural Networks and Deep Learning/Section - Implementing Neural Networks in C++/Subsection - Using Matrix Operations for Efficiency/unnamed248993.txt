#include <Eigen/Dense>
#include <iostream>

int main() {
    Eigen::Matrix3d A;
    A << 1, 2, 3,
         4, 5, 6,
         7, 8, 9;

    Eigen::Matrix3d B;
    B << 9, 8, 7,
         6, 5, 4,
         3, 2, 1;

    Eigen::Matrix3d C = A * B;

    std::cout << C << std::endl;
    return 0;
}
