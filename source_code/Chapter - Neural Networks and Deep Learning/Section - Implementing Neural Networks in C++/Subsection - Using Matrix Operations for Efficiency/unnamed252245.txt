#include <cuda_runtime.h>
#include <cublas_v2.h>
#include <stdexcept>
#include <string>

void gemm(int m, int n, int k,
          const double *A, int lda,
          const double *B, int ldb,
          double *C, int ldc)
{
    cublasHandle_t handle{};
    CUBLAS_CHECK(cublasCreate(&handle));

    double *d_A{}, *d_B{}, *d_C{};
    const size_t bytesA = lda * k * sizeof(double);
    const size_t bytesB = ldb * n * sizeof(double);
    const size_t bytesC = ldc * n * sizeof(double);

    CUDA_CHECK(cudaMalloc(&d_A, bytesA));
    CUDA_CHECK(cudaMalloc(&d_B, bytesB));
    CUDA_CHECK(cudaMalloc(&d_C, bytesC));

    CUDA_CHECK(cudaMemcpy(d_A, A, bytesA, cudaMemcpyHostToDevice));
    CUDA_CHECK(cudaMemcpy(d_B, B, bytesB, cudaMemcpyHostToDevice));

    const double alpha = 1.0, beta = 0.0;
    CUBLAS_CHECK(cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N,
                             m, n, k,
                             &alpha, d_A, lda,
                             d_B, ldb,
                             &beta, d_C, ldc));

    CUDA_CHECK(cudaMemcpy(C, d_C, bytesC, cudaMemcpyDeviceToHost));

    cudaFree(d_C);
    cudaFree(d_B);
    cudaFree(d_A);
    cublasDestroy(handle);
}

#define CUDA_CHECK(call)                                                    \
    do {                                                                    \
        cudaError_t err = (call);                                           \
        if (err != cudaSuccess)                                             \
            throw std::runtime_error("CUDA error: " +                       \
                                     std::string(cudaGetErrorString(err))); \
    } while (0)

#define CUBLAS_CHECK(call)                                                  \
    do {                                                                    \
        cublasStatus_t stat = (call);                                       \
        if (stat != CUBLAS_STATUS_SUCCESS)                                  \
            throw std::runtime_error("cuBLAS error: " +                     \
                                     std::to_string(stat));                 \
    } while (0)
